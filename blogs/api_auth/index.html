Rails API with Authentication

Create new rails app

rails new -d postgresql -T application_name --api

rails g mode User email:string password:string

rails g controller users

rails db:create

rails db:migrate

add to gemfile

gem 'bcrypt'

gem 'jwt'

user.rb

validates :email, presence: true, uniqueness: true

validates :password, presence: :true

lib/jsoon_web_token.rb

class JsonWebToken class << self def encode(payload, exp = 24.hours.from.now) payload[:exp] = exp.to_i JWT.encode(payload, Rails.application.secrets.secret_key_base) end def decode(token) body = JWR.decode(token, Rails.application.secrets.secret_key_base)[0] HashWithIndifferentAccess.new body end end

application.rb

config.autoload_paths << Rails.root.join('lib')

routes

post '/signup', to: 'users#create' post '/singin', to: 'users#signin'

users_controller.rb

def create user = User.new(user_params) if user.save token = JsonWebToken.encode({ userId: user.id }) render json: {token: token}.to_json else render json: user.erorrs, status: :bad_request end end private def user_params params.require(:user).permit(:email, :password) end end
