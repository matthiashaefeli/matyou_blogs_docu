# Ruby add only to array if element does not exist in array

```
2.7.2 :033 > array = [1,2,3,4]
 => [1, 2, 3, 4]
2.7.2 :034 > array |= [5]
 => [1, 2, 3, 4, 5]
2.7.2 :035 > array
 => [1, 2, 3, 4, 5]
2.7.2 :036 > array |= [5]
 => [1, 2, 3, 4, 5]
2.7.2 :037 > array
 => [1, 2, 3, 4, 5]
```

```
2.7.2 :008 > array = [1,2,3,4]
 => [1, 2, 3, 4]
2.7.2 :009 > array.push(5) unless array.include?(5)
 => [1, 2, 3, 4, 5]
2.7.2 :010 > array
 => [1, 2, 3, 4, 5]
2.7.2 :011 > array.push(5) unless array.include?(5)
 => nil
2.7.2 :012 > array
 => [1, 2, 3, 4, 5]
```
